## Merge Sorted Array
Hint: Two Pointer

``` java 
class Solution {
    // Two pointer starting backwards in the larger array
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        if (m == 0) {
            for (int i  = 0; i < n; i++) {
                nums1[i] = nums2[i];
            }
        }
        else if (n != 0) {
            int p1 = m - 1;
            int p2 = n - 1;
            int p3 = nums1.length - 1;

            while ((p1 >= 0) && (p2 >= 0)) {
                if (nums1[p1] < nums2[p2]) {
                    nums1[p3] = nums2[p2];
                    p2--;
                }
                else {
                    nums1[p3] = nums1[p1];
                    p1--;
                }
                p3--;
            }

            if (p2 >= 0) {
                while (p2 >= 0) {
                    nums1[p3] = nums2[p2];
                    p2--;
                    p3--;
                }
            }

            if (p1 >= 0) {
                while (p1 >= 0) {
                    nums1[p3] = nums1[p1];
                    p1--;
                    p3--;
                }
            }

        }
    }
}
